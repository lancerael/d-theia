    import Utilities from './Utilities';
import Colors from './Colors';

/**
* The DataOps object is is a class containing reusable static methods to be operated on data
*
* @class DataOps
* @static
*/
export default class DataOps {

  /**
  * Returns a sample selection from an array (minmum length 3).
  *
  * @method sliceSampleData
  * @static
  * @param {Array} chartData the complete data set
  * @param {Integer} maxLength max length of sample
  * @return {Array} sliced section of data
  */
  static sliceSampleData(chartData, maxLength = 50) {
    const iStart = Utilities.getRandomInteger(0, chartData.length - maxLength);
    const iEnd = iStart + Utilities.getRandomInteger(3, maxLength);
    return chartData.slice(iStart, iEnd);
  }

  /**
  * Returns random data and config for demo purposes.
  *
  * @method getRandomChart
  * @static
  * @return {JSON} JSON style data structure
  */
  static getRandomData() {
    const length = Utilities.getRandomInteger(3, 10);
    const groupSize = Utilities.getRandomInteger(1, 4);
    const rangeLow = Utilities.getRandomInteger(1, 50);
    const rangeHigh = Utilities.getRandomInteger(rangeLow, 200);
    const chartParams = {};
    chartParams.chartConfig = {
      title: 'Random Chart Data',
      axisLabels: ['Y Axis', 'X Axis'],
      itemValues: (() => {
        const itemValues = [];
        let counter = groupSize;
        while (counter--) {
          itemValues.push({ name: `Type ${groupSize - counter}` });
        }
        return itemValues;
      })(),
      doTrim: true
    };
    chartParams.chartConfig.itemValues = DataOps.addColoursToConfig(chartParams.chartConfig.itemValues, false);
    chartParams.chartData = [];
    let counter = length;
    while (counter--) {
      chartParams.chartData.push({
        itemLabel: `Item ${length - counter}`,
        itemValues: (() => {
          const itemValues = [];
          let subCounter = groupSize;
          while (subCounter--) {
            itemValues.push(Utilities.getRandomInteger(rangeLow, rangeHigh));
          }
          return itemValues;
        })()
      });
    }
    return chartParams;
  }

  /**
  * Maps custom data keys into standard structure.
  *
  * @method transformDataKeys
  * @static
  * @param {JSON} chartConfig the JSON style config object
  * @param {Array} chartData the chart data to be transformed
  */
  static transformDataKeys(chartConfig, chartData) {
    return chartData.map((item) => {
      if (chartConfig.itemValues && !item.itemValues) {
        item.itemValues = chartConfig.itemValues.map(value => parseInt(item[value.key]));
      }
      if (item.itemValues) {
        item.itemValues = item.itemValues.map((value) => {
          const sanitisedValue = parseInt(value);
          return isNaN(sanitisedValue) ? 0 : sanitisedValue;
        });
      }
      if (chartConfig.axisKeys && !item.itemLabel) {
        item.itemLabel = item[chartConfig.axisKeys[0]];
      }
      return item;
    });
  }

  /**
  * Adds colour values to the config array of key values
  *
  * @method addColoursToConfig
  * @static
  * @param {Array} itemValues the config values needing colours
  * @param {Boolean} bAddRgb add RGB object or not?
  * @return {Array} A modified version of itemValues containing colours
  */
  static addColoursToConfig(itemValues, bAddRgb = true) {
    return itemValues.map((value) => {
      if (!value.color) {
        value.color = Colors.getRandomColor().hex();
      }
      if (bAddRgb) {
        value.color = Colors.convertHexToRgb(value.color);
      }
      return value;
    });
  }

}
