    import { axisBottom, axisLeft } from 'd3-axis';
import Utilities from './Utilities';

/**
* The Axis object is used to instantiate x and y axes, as well as labels
*
* @class Axis
* @constructor
*/
export default class Axis {

  /**
  * d3 object for axis container
  *
  * @property d3Container
  * @type {Object}
  */
  d3Container;

  /**
  * Amount to truncate axis labels to
  *
  * @property truncateSize
  * @type {Object}
  */
  truncateSize;

  /**
  * Collection of axis labels
  *
  * @property axisLabels
  * @type {Array}
  */
  axisLabels;

  /**
  * Scale object for the x axis
  *
  * @property scaleX
  * @type {Object}
  */
  scaleX;

  /**
  * Scale object for the y axis
  *
  * @property scaleY
  * @type {Object}
  */
  scaleY;

  /**
  * Chart's tooltip object
  *
  * @property tooltip
  * @type {Object}
  */
  tooltip;

  /**
  * The current calculated width of the chart
  *
  * @property width
  * @type {Number}
  */
  width;

  /**
  * The current calculated height of the chart
  *
  * @property height
  * @type {Number}
  */
  height;

  /**
  * The padding for the chart within the container
  *
  * @property padding
  * @type {Object}
  */
  padding;

  /**
  * Constructor function which sets up the local object.
  *
  * @method constructor
  * @param {Object} chartParams configuration parameters object
  * @throws {Error} invalid parameters
  */
  constructor(chartParams) {
    if (chartParams.d3Container) {
      Object.assign(this, chartParams);
    } else {
      throw new Error('Incorrect parameters provided to Axis constructor.');
    }
  }

  /**
  * Master render to call all rendering methods
  *
  * @method render
  * @chainable
  */
  render() {
    this.renderAxisX();
    this.renderAxisY();
    this.renderLabels();
    return this;
  }

  /**
  * Render only the x axis
  *
  * @method renderAxisX
  */
  renderAxisX() {
    this.d3Container.selectAll('g.x-axis').remove();
    this.d3Container.append('g')
        .attr('class', 'x-axis')
        .call(axisBottom(this.scaleX))
        .attr('transform', `translate(${this.padding.l},${this.height})`)
        .selectAll('text')
          .attr('x', -5)
          .attr('y', 6)
          .attr('transform', 'rotate(310)')
          .attr('class', 'x-labels')
          .text(d => Utilities.truncateString(d, this.truncateSize))
          .style('text-anchor', 'end')
          .on('mousemove', (d) => {
            if (this.tooltip && d.length > this.truncateSize) {
              this.tooltip.ping(`<strong>${d}</strong>`);
            }
          });
  }

  /**
  * Render only the y axis
  *
  * @method renderAxisY
  */
  renderAxisY() {
    this.d3Container.selectAll('g.y-axis').remove();
    this.d3Container.append('g')
        .attr('class', 'y-axis')
        .call(axisLeft(this.scaleY))
        .attr('transform', `translate(${this.padding.l},0)`)
        .selectAll('.y-axis .tick line')
          .attr('x2', () => this.width);
  }

  /**
  * Render only the labels
  *
  * @method renderLabels
  */
  renderLabels() {
    this.d3Container.selectAll('text.labels').remove();
    this.d3Container.append('text')
        .attr('class', 'labels')
        .attr('x', (this.height / -2) - (this.padding.t / 2))// - ((this.chartConfig.axisLabels[0].length / 2) * 7))
        .attr('y', 10)
        .attr('transform', 'rotate(-90)')
        .attr('text-anchor', 'middle')
        .text(this.axisLabels[0]);
    this.d3Container.append('text')
        .attr('class', 'labels')
        .attr('x', ((this.width + this.padding.l + this.padding.r) / 2))// - ((this.axisLabels[1].length / 2) * 2))
        .attr('y', this.height + (this.padding.b - 5))
        .attr('text-anchor', 'middle')
        .text(this.axisLabels[1]);
  }

}

module.exports = Axis;
