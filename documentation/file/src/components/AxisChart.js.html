    src/components/AxisChart.js

    import { min, max } from 'd3-array';
import { select } from 'd3-selection';
import { scaleLinear, scaleBand } from 'd3-scale';
import Chart from './Chart';
import Key from './Key';
import Axis from './Axis';

/**
* Create AxisChart from the supplied data, based on the supplied JSON config.
*
* @class AxisChart
* @extends {Chart}
* @constructor
*/
export default class AxisChart extends Chart {

  /**
  * The d3 object for the chart's group
  *
  * @property d3ChartGroup
  * @type {Object}
  */
  d3ChartGroup;

  /**
  * The d3 object for the chart's axis
  *
  * @property axisGroup
  * @type {Object}
  */
  axisGroup;

  /**
  * The chart's axis object
  *
  * @property axis
  * @type {Object}
  */
  axis;

  /**
  * The chart's key object
  *
  * @property key
  * @type {Object}
  */
  key;

  /**
  * Constructor function supersedes parent class.
  *
  * @method constructor
  * @param {Object} chartParams same as Chart
  */
  constructor(chartParams = {}) {
    super(chartParams);
    this.padding = { l: 45, r: 15, t: 25, b: 90 };
    this.scaleY = scaleLinear();
    this.scaleX = scaleBand().padding(0.2);
  }

  /**
  * Supersede the parent method to update local scaling objects
  *
  * @method setDimensions
  */
  setDimensions() {
    super.setDimensions();
    const doTrim = this.chartConfig.doTrim;
    let minValue = doTrim ? min(this.chartData, d => min(d.itemValues)) : 0;
    let maxValue = max(this.chartData, d => max(d.itemValues));
    const dataSlice = Math.ceil(maxValue / 15);
    if (doTrim) {
      const iLowerSection = minValue > dataSlice ? minValue - dataSlice : 0;
      minValue = minValue > 0 ? iLowerSection : minValue;
      minValue = minValue < 0 ? minValue - dataSlice : minValue;
      maxValue += dataSlice;
    }
    this.scaleX
      .domain(this.chartData.map(d => d.itemLabel))
      .range([0, this.innerWidth]);
    this.scaleY
      .domain([minValue, maxValue])
      .range([this.innerHeight, this.padding.t]);
    this.minValue = minValue;
    this.maxValue = maxValue;
    this.dataSlice = dataSlice;
  }

  /**
  * Render the chart including axes and labels
  *
  * @method renderChart
  */
  renderChart() {
    const { axisLabels, truncateSize = 15 } = this.chartConfig;
    const { innerWidth, innerHeight, scaleX, scaleY, padding } = this;

    // Add chart scale axes
    this.axisGroup = this.axisGroup || this.d3Svg.append('g').attr('class', 'axes-g');
    this.axis = new Axis({
      d3Container: this.axisGroup,
      truncateSize,
      axisLabels,
      scaleX,
      scaleY,
      padding,
      width: innerWidth,
      height: innerHeight,
      tooltip: this.tooltip }).render();

    // Add chart container group
    this.d3ChartGroup = this.d3ChartGroup || this.d3Svg.append('g').attr('transform', `translate(${this.padding.l}, 0)`);

    // Render the key for the data
    this.key = new Key({
      d3Container: select(this.svg),
      itemValues: this.chartConfig.itemValues,
      offsetX: (this.innerWidth / 2) + this.padding.l,
      offsetY: this.height - 20
    }).render();
  }

}

module.exports = AxisChart;
